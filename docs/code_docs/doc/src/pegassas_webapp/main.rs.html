/// User login information
#[derive(Deserialize)]
struct LoginForm {
    username: String,
    password: String
}

/// Probe request frequency information
#[derive(Deserialize)]
struct ProbeForm {
    probe_name: String,
    data: String,
    frequency: i32
}

/// View configuration update form information
#[derive(Deserialize, Serialize)]
struct ViewForm {
    view_tab : i32,
    container_index : i32,
    y_value : String,
    range_start : DateTime<Utc>,
    range_finish : DateTime<Utc>,
    where_id : String,
    graph_type : String,
    tab_name : Option<String>
}

#[derive(Deserialize)]
struct RegistrationForm{
    username: String,
    password: String,
    email: String,
}


async fn registration(form: web::Form<RegistrationForm>) -> HttpResponse{
    let (client, connection) = tokio_postgres::connect("host=localhost user=alikhan password=thisisapassword dbname=pegassas", NoTls).await.unwrap();
    tokio::spawn( async {
        if let Err(e) = connection.await{
            eprintln!("Error {}", e);
        }
    });
    match check_username(&form.username).await{
        Ok(true) => {
            return HttpResponse::BadRequest().body("Username already exists");
        },
        Ok(false) => (),
        Err(e) => {
            error!("Error producing password hash: {}", e);
            return HttpResponse::build(StatusCode::INTERNAL_SERVER_ERROR).body("Password hash Error");
        }
    };
    let pass_hash:String = match bcrypt::hash(&form.password, 12){
        Ok(p) => p,
        Err(e) => {
            error!("Query failed requesting the user data: {}", e);
            return HttpResponse::build(StatusCode::INTERNAL_SERVER_ERROR).body("Database access error");
        }
    };
    let response = match client.execute(
        "INSERT INTO user_account 
        values(DEFAULT, $1, $2, $3);",
        &[&form.username, &pass_hash, &form.email],
    ).await{
        Ok(r) => r,
        Err(e) => {
            error!("Query failed inserting the user data: {}", e);
            return HttpResponse::build(StatusCode::INTERNAL_SERVER_ERROR).body("Query Error");
        }
    };
    if response == 1{
        HttpResponse::Ok()
            .content_type("text/plain")
            .body(format!("Update successful"))
    }else{
        HttpResponse::Ok()
            .content_type("text/plain")
            .body(format!("Error"))
    }
}

async fn check_username(username:&String) -> Result<bool, tokio_postgres::error::Error>{
    let (client, connection) = tokio_postgres::connect("host=localhost user=alikhan password=thisisapassword dbname=pegassas", NoTls).await.unwrap();
    tokio::spawn( async {
        if let Err(e) = connection.await{
            eprintln!("Error {}", e);
        }
    });
    let response = match client.query(
        "Select user_id from user_account where username = $1;",
        &[&username],
    ).await{
        Ok(r) => r,
        Err(e) => {
            error!("Query failed requesting the user data: {}", e);
            return Err(e);
        }
    };

}

/// Request view setting configuration from database and return it in a Json Http response
async fn view_config_request(session:Session) -> HttpResponse{
    let (client, connection) = tokio_postgres::connect("host=localhost user=alikhan password=thisisapassword dbname=pegassas", NoTls).await.unwrap();
    tokio::spawn( async {
        if let Err(e) = connection.await{
            eprintln!("Error {}", e);
        }
    });
    let user_id: i32 = match session.get("user_id"){
        Ok(Some(user_id)) => user_id,
        Ok(None) => {
            error!("Withdrawing user id from the session produced error");
            return HttpResponse::build(StatusCode::INTERNAL_SERVER_ERROR).body("Session Error");
        }
        Err(e) => {
            error!("Withdrawing user id from the session produced error: {}", e);
            return HttpResponse::build(StatusCode::INTERNAL_SERVER_ERROR).body("Session Error");
        }
    };
    let response = client.query(
        "Select * from view_configuration where user_id = $1", &[&user_id]
    ).await.unwrap();
    let mut views = Vec::new();
    for row in response {
        if row.is_empty() {continue;}
        let view = ViewForm {
            view_tab : row.get("view_tab"),
            container_index : row.get("container_index"),
            y_value : row.get("y_value"),
            range_start : row.get("range_start"),
            range_finish : row.get("range_finish"),
            where_id : row.get("where_id"),
            graph_type : row.get("graph_type"),
            tab_name : row.get("tab_name")
        };
        views.push(view);
    }
    web::HttpResponse::Ok().json(views)
}

/// Request probe configuration from database and return it as a Json HttpResponse
async fn request_probe_config() -> HttpResponse{
    let (client, connection) = match tokio_postgres::connect("host=localhost user=alikhan password=thisisapassword dbname=pegassas", NoTls).await{
        Ok(c) => c,
        Err(e) => {
            error!("PostgreSQL connection error: {}", e);
            return HttpResponse::build(StatusCode::INTERNAL_SERVER_ERROR).body("PostgreSQL connection error");
        }
    };
    tokio::spawn( async {
        if let Err(e) = connection.await{
            error!("PostgreSQL connection error: {}", e);
        }
    });
    let query = "select array_to_json(array_agg(row_to_json(t)))
    from (
        Select mac_address, request_interval from probe_config
    ) t";
    let response = match client.query(query, &[]).await{
        Ok(r) => r,
        Err(e) => {
            error!("PostgreSQL query: {} produced error: {}", query, e);
            return HttpResponse::build(StatusCode::INTERNAL_SERVER_ERROR).body("PostgreSQL Query Error");
        }
    };
    let responseLine:serde_json::Value = response[0].get(0);
    web::HttpResponse::Ok().json(responseLine)
}

/// Change request interval for a specific probe
async fn probe_config(session: Session, mut form: web::Form<ProbeForm>) -> HttpResponse{
    let (client, connection) = tokio_postgres::connect("host=localhost user=alikhan password=thisisapassword dbname=pegassas", NoTls).await.unwrap();
    tokio::spawn( async {
        if let Err(e) = connection.await{
            error!("PostgreSQL connection error: {}", e);
        }
    });
    if form.data == "Off" || form.data == "off"{
        form.frequency = 0
    }
    let mac_address = match eui48::MacAddress::parse_str(form.probe_name.as_str()){
        Ok(m) => m,
        Err(e) => return HttpResponse::BadRequest().body("Bad Mac Address format")
    };
    let response = client.execute(
        "Update probe_config p set request_interval = $1 where p.mac_address = $2;",
        &[&form.frequency , &mac_address],
    ).await.unwrap();
    if response == 1{
        HttpResponse::Ok()
            .content_type("text/plain")
            .body(format!("Update successful"))
    }else{
        HttpResponse::Ok()
            .content_type("text/plain")
            .body(format!("Error"))
    }
}

/// Verify password string input against the encrypted password held in database
pub fn pass_verify(db_password :String, password: String) -> bool{
        match bcrypt::verify(password, db_password.as_str()){
            Ok(b) => b,
            Err(_) => {
                println!("bcrypt error");
                false
            }
        }  
}

/// Direct to homepage
pub fn to_home() -> HttpResponse {
    HttpResponse::Found().header(LOCATION, "/home.html").finish()
  }

/// Check active user in current session
pub fn is_signed_in(session: &Session) -> bool {
    match session.get::<String>("user"){
        Ok(Some(_)) => true,
        Err(e) => {
            error!("Unable to get user from session: {:?}", e);
            false
        }
        _ => false
    }
}


#[actix_web::main]
async fn main() -> std::io::Result<()> {
    setup_logger();
    HttpServer::new(move || {
        let app = App::new()
        .wrap(CookieSession::signed(&[0; 32]).secure(true))
        .service(
            web::resource("/login.html")
                .route(web::post().to(login))
        )
        .service(
            web::resource("dbapi/cell")
                .route(web::get().to(dbapi::request_cell))
        )
        .service(
            web::resource("setprobes")
                .route(web::post().to(probe_config))
        )
        .service(
            web::resource("getprobes")
                .route(web::get().to(request_probe_config))
        )
        .service(
            web::resource("dbapi/battery")
                .route(web::get().to(dbapi::request_battery))
        )
        .service(
            web::resource("dbapi/inverter")
                .route(web::get().to(dbapi::request_inverter))
        )
        .service(
            web::resource("dbapi/grid")
                .route(web::get().to(dbapi::request_grid))
        )
        .service(
            web::resource("dbapi/house")
                .route(web::get().to(dbapi::request_house))
        )
        .service(
            web::resource("/view_config")
                .route(web::post().to(view_config))
        )
        .service(
            web::resource("/get_view_config")
                .route(web::get().to(view_config_request))
        )
        .service(
            web::resource("/registration")
                .route(web::post().to(registration))
        );
        let mut is_dev = false;
        for (key, value) in std::env::vars_os(){
            if key == "PEGASSAS_WEBAPP_ENV" {
                if value == "DEV" {
                    is_dev = true;
                }
                break;
            }
        }
        if is_dev {
            info!("Running in development mode");
            app.service(fs::Files::new("/", "./static/").index_file("login.html"))
        }else{
            app.service(fs::Files::new("/", "/usr/share/pegassas_webapp/static/").index_file("login.html"))
        }
        
    })
        .bind("127.0.0.1:8089")?
        .run()
        .await
    

        // Note from Max
        
        // Each row is a Form that uses the POST method to /config.html (when u press submit)
        // The variables are below:
        // id="Probe_X"
        // data=on (This is optional, if off there wont be a data variable)
        // frequency=4 (This is optional, if data is disabled there wont be a frequency variable)

        // See example urls below

        // Probe 1 data is on and set to 3
        // http://127.0.0.1:8082/config.html?id=Probe_1&data=on&frequency=3
        
        // Probe 3 data is off
        // http://127.0.0.1:8082/config.html?id=Probe_3
}
